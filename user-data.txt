#!/bin/bash
usr=ec2-user
yum update -y

# Remove previous build
rm /home/${usr}/*.jar

# Copy new build from s3 bucket to user home
aws s3 cp s3://auto-aid-adviser-bucket/build/auto-aid-adviser-main-boot.jar /home/${usr}/

cd /home/${usr}/

# start spring-boot app in dev mode
java -jar -Dspring.profiles.active=dev auto-aid-adviser-main-boot.jar

#This magic line (only works with sudo) redirects http from external 80 to internal 8080:
sudo iptables -t nat -I PREROUTING -p tcp --dport 80 -j REDIRECT --to-ports 8080
--//
